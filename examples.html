fnEnumColumn for DataTables

A plugin for DataTables that fills any column with row numbers.

Examples

// Get dataTable object (assume it's already initialized):
var dt = $('table#example').dataTables();

// The first column will display formated row numbers
dt.fnEnumColumn({format: 'Row: %d'});

// The second column will display 'caclulated' data
dt.fnEnumColumn({
    columnNumber: 2,
    format: function (rowNum, aData){ return rowNum * 3;}
});
